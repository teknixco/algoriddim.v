void CheckForOpenMAConditions()
  {
   if(
      baseConditionsForBuyOrder()
      && MAConditionsForBuyOrder()
      && shouldAllowBuyFromRSI[M1]
      && isThisBarBULLISH()
      && getRSI(M1) < 55
      && getMA(M1, _50) < getLowestPrice(1, 1)
      && getMA(M5, _20) < getLowestPrice(5, 1)
   )
     {
      placeBuyOrder(orderLots, "BuyMA", 100);
     }
  }
